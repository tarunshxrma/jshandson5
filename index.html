Q1. Count Characters


const countCharacters = (S) => 
{
  let upper=0;
  let lower=0;
  for(let i=0;i< S.length;i++)
  {
    let ch = S.charAt(i);
    if(ch === 'A')
    upper++;
    else if (ch==='D'){
      lower++;
    }
  }
  return [upper,lower];
};

var Count_Vowels= (S) => 
{
  sum=0;
  let vowel=[ 'A','E','I','O','U','a','e','i','o','u'];
  for(let i=0;i< S.length;i++){
    for(let j=0;j< vowel.length;j++){
      if(S[i]===vowel[j]){
        sum+=1;
      }
    }
  }
  return sum;
      
};

Q6.Find the Winner


 
var Game_Winner = (S) => 
{
  let countOfA=0;
  let countOfD=0;
  for(i=0;i< S.length;i++){
    if(S[i]==="A"){
      countOfA++;
    }
    else if(S[i]==="D"){
      countOfD++;
    }
  }
  if(countOfA===countOfD){
    return "Draw";
  }
  else if(countOfA>countOfD){
    return "Aditya";
  }
  else if(countOfD>countOfA){
    return "Danish";
  }
  
}

Q8.Plaindrome Check


 
var Palin_Check = (str) =>
{
  const len = str.length;
  for(let i=0;i< len/2;i++){
    if(str[i]!==str[len-1-i]){
      return 'False';
    }
  }
  return 'True';
	 
};

Q9.Reverse the String


var Reverse_String = (str) => 
{
  let newString="";
  for(let i=str.length-1;i>=0;i--){
    newString += str[i];
  }
  return newString;
     
};
